### hocus pokus

'a = [in, van, maar, op, aan]
'a + le = au
'a + la = 'a la
'a +'l = 'a 'l
'a + les = aux

de = [van, uit]
de + le = du
de + la = de la
de + 'l = de 'l
de + les = des
